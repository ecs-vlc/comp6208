\begin{question}{10}
  \begin{qparts}
    \qpart[5]{Newton's method is not guaranteed to converge
      if we start too far from $x^{*}$.  The Banach fixed-point
      theorem guarantees that if we apply a contraction mapping\,
      $\mathcal{T}$ from a metric space, $\mathcal{V}$, to itself applied multiple
      times will reach a unique fixed-point.  The mapping
      $T:\mathcal{V}\rightarrow \mathcal{V}$ is a contraction mapping if
      \begin{align*}
        d(T(\bm{x}) , T(\bm{y})) \leq d(\bm{x}- \bm{y})
      \end{align*}
      for all $\bm{x}$ and $\bm{y}$.

      We consider the mapping $f:\mathbb{R}\rightarrow\mathbb{R}$
      with a metric $d(x,y) = |x-y|$. Let $\epsilon>0$ and let
      $x=y+\epsilon$ and assume that $f(x)$ is suitable well behaved
      that it can be approximated by the Taylor expansion
      $f(y+\epsilon) = f(y) + \epsilon f'(y) + O(\epsilon^2)$ obtain
      a condition for $f(x)$ to be a contraction mapping in terms of
      the derivative $f'(x)$.}{\lines{10}}
    \begin{answer}
      Substituting in $x= y+\epsilon$ into  $|f(x) - f(y)|$
      we find
      \begin{align*}
        |f(x) - f(y)|
        &= |f(y + \epsilon) -f(y)| = |\epsilon\, f'(y) + O(\epsilon^2)|
        \\
        &\leq |\epsilon\, f'(y)| + | O(\epsilon^2)|
          = \epsilon \, |f'(y)| +\epsilon^2 \, | O(1)|
      \end{align*}
      where we $|x+y| <|x| +|y|$ (this is the triangular inequality in
      this metric space).  For this to be true in the limit
      $\epsilon\rightarrow 0$ we require $ |f(x) - f(y)| \leq
      \epsilon \, |f'(y)|$, but $\epsilon = x-y$ so that
      \begin{align*}
         |f(x) - f(y)| \leq |f'(y)|\, |x-y|,
      \end{align*}
      Thus a sufficient condition for $f(x)$ to be contraction
      mapping is that $f'(x)<1$ for all $x$.  This is also a
      necessary condition as
      \begin{align*}
         |f(x) - f(y)|
        &=  |\epsilon\, f'(y) + O(\epsilon^2)| \geq
          |\epsilon\, f'(y)| - |O(\epsilon^2)| \
        \\
        &=  \epsilon\, |f'(y)| - \epsilon^2 \,|O(1)|           
      \end{align*}
      If $|f'(y)|\geq 1$ at any point $y$ then in the limit $\epsilon
      \rightarrow 0$ the mapping would not be a contraction mapping at
      that point.
    \end{answer}
  \end{qparts}
\end{question}
